Android Studio中调试native代码
Debug Android Native
发布日期:   2019-09-23

1. 背景

1.1 技术背景

随着减少多端开发成本，规避多端差异，业务采用跨端的C++开发，本文主要介绍在Android调试native代码的技巧，抛砖引玉。

2. 调试机型

2.1 有root权限的手机

2.1.1 命令行中敲如上命令，将本地编译好的so文件push到手淘的安装路径下
adb push libmesssagesdkwrapper.so /sdcard/tmp/
adb shell
// inside adb shell
su
cp /sdcard/tmp/libmessagesdkwrapper.so /data/app/com.taobao.taobao-2/lib/arm/

可以使用附件中的devtool.py脚本简化操作：

devtool.py pushlib libmessagesdkwrapper.so com.taobao.taobao

脚本会过滤出子目录下所有的aar文件，筛选出合适的文件解压并上传so文件到app的安装目录。
(脚本通过找到libC++_shared.lib来获得lib目录，如果有更通用的方案欢迎拍砖)

2.1.2 然后操作Android Studio，Run->Attach Debugger to Android Process。
2.1.3 选择Native调试器，选中手淘的进程，点击OK等待附加完成。

2.2 没有root权限的手机

主要原理就是将现有的apk解压，然后替换里面的so文件，重新打包签名。

2.2.1 工具下载

Java环境: 如果本地没有安装Java环境，又不想安装Java环境，可以直接使用Android Studio内部集成的jre环境，配置环境变量JAVA_HOME=C:\Program Files\Android\Android Studio\jre，在PATH中增加%JAVA_HOME%/bin
ApkTool助手：(其实可以用ApkTool命令行直接操作的，这里偷懒使用别人已经封装好的UI程序)
下载链接: https://pan.baidu.com/s/17Id9xxn0NdFdTOhfrM_T8A 密码:d590

2.2.2 更改ApkTool助手配置:

由于手淘中集成了安全保镖组件，保镖会校验当前apk的签名，如果不合法就不能进行加签等操作，所以我们需要依赖手淘的签名。
手淘的签名因为涉及到安全问题，请谨慎操作！建议修改Tool\Config.ini文件，keystore节点下的keystore指向淘宝的debug.keystore文件。

2.2.3 开始反编译apk文件

反编译的过程因为涉及到dex->smali，整个进程比较慢。

完成之后，将编译好的so文件替换导出目录 lib\armeabi-v7a

2.2.4 打包apk+签名

选择刚才导出的目录，点击回编译+签名

整个过程比较缓慢，后续再打包时会检测smali文件是否有变更，操作就会快很多了

2.2.5 将apk推送到手机安装
adb install -r taobao-android-debug_mod_singed.apk

安装之后，调试方法参照2.1.2 + 2.1.3

3. 附加进程时提示Exception in plugin Android NDK Support的解决办法

因为Andoid NDK Support在开启远端lldb server时候，需要调用run-as {packagename} lldb-server，这样才有权限attach目标进程，这里会使用当前工程的包名，如果当前包名非com.taobao.taobao，会导致lldb server不能成功被创建或者无法attach到taobao的进程(调试java的模块就没有这个烦恼)。简单的方案是在工程中新建一个安卓的module，包名改为手淘包名。如果项目中存在demo工程，直接将app/build.gradle中的包名改为手淘的包名即可。

#!/usr/bin/env python
# encoding: utf-8
import argparse
import os
import re
import shutil
import subprocess
import sys
import zipfile
import hashlib

""" Toos for developer
Usage:
  # push lib files on *ROOT* (support su) device
  devtool pushlib libmessagesdkwrapper.so com.taobao.taobao
"""

# 目前已知MEIZU自带的su命令使用c作为key。规范应该使用su -c ""
BRAND_UPPER_SU_COMMAND_C = [
  "MEIZU",
]

CMD_ENCODING = sys.stdout.encoding if sys.stdout.encoding is not None else "gbk"

def choice(prompt, choices=["y", "n"]):
  user_input = ""
  while user_input not in choices:
    if isinstance(prompt, unicode):
      user_input = raw_input(prompt.encode(CMD_ENCODING, "ignore"))
    else:
      user_input = raw_input(prompt)
  return user_input


def md5sum(filepath):
  if os.path.isfile(filepath):
    with open(filepath, "rb") as fp:
      contents = fp.read()
      return (hashlib.md5(contents).hexdigest())
  return None


class AdbShell(object):
  def __init__(self, adb_path, use_su = False):
    self.adb_path = adb_path
    self.use_su = use_su
    if self.use_su:
      self.su_command = self._get_su_command()

  def _get_su_command(self):
    brand = subprocess.check_output([self.adb_path, "shell", "getprop", "ro.product.brand"])
    return "c" if brand.strip().upper() in BRAND_UPPER_SU_COMMAND_C else "-c"

  def check_call(self, args):
    if not isinstance(args, list):
      raise Exception("unsupport args!")
    if self.use_su:
      return subprocess.check_call([self.adb_path, "shell", "su", self.su_command,
        "%s" % (" ".join([v.replace('"', "'") for v in args]))])
    else:
      return subprocess.check_call([self.adb_path, "shell"] + args)

  def check_output(self, args):
    if not isinstance(args, list):
      raise Exception("unsupport args!")
    if self.use_su:
      return subprocess.check_output([self.adb_path, "shell", "su", self.su_command,
        "%s" % (" ".join([v.replace('"', "'") for v in args]))])
    else:
      return subprocess.check_output([self.adb_path, "shell"] + args)


def _join_path(path, file):
  path = os.path.join(path, file)
  return path.replace("\\", "/")


def _get_project_output():
  return os.path.dirname(os.path.abspath(__file__))


def _find_package_pid(adb_path, package):
  """Find the pid of the application process."""
  ps_output = subprocess.check_output([adb_path, 'shell', 'ps'])
  ps_match = re.search(r'^\S+\s+(\d+).*\s%s[^:]' % package, ps_output, re.MULTILINE)
  if not ps_match:
    raise Exception('Unable to find pid for package %s on device' % package)
  return int(ps_match.group(1))


def _get_app_path(adb_path, package):
  package_pid = _find_package_pid(adb_path, package)
  proc_maps = subprocess.check_output([adb_path, 'shell', 'run-as', package, 'cat', '/proc/%d/maps' % package_pid])
  match = re.search(r'(/data/\S*?libc\+\+_shared\.so)', proc_maps)
  if match is None:
    raise Exception('Unable to app path on proc_maps(%d)' % (package_pid))
  return os.path.dirname(match.group(0))


def _get_app_abi(adb_path, package):
  abi_output = subprocess.check_output([adb_path, 'shell', 'getprop', 'ro.product.cpu.abi']).strip()
  if abi_output.startswith('arm'):
    package_pid = _find_package_pid(adb_path, package)
    proc_maps = subprocess.check_output([adb_path, 'shell', 'run-as', package, 'cat', '/proc/%d/maps' % package_pid])
    if proc_maps.find("/system/bin/app_process64") != -1:
      return "arm64-v8a"
    else:
      return "armeabi-v7a"
  return abi_output


class PushLib(object):
  DEVICE_TMP_PATH = '/data/local/tmp'

  def add_subparser(self, subparser):
    parser = subparser.add_parser('pushlib', help="push lib to package lib directory")
    parser.add_argument('lib', type=str)
    parser.add_argument('package', type=str)
    parser.set_defaults(func=self.run)

  def _find_output_aar(self):
    file_list = []
    for root, _, files in os.walk("."):
      root = root.replace("\\", "/")
      if root.endswith("build/outputs/aar"):
        for v in files:
          if os.path.splitext(v)[1] == ".aar":
            file_list.append(_join_path(root, v))
    return file_list

  def _find_lib(self, aar, abi, lib):
    zip_file = zipfile.ZipFile(aar)
    info = zip_file.getinfo("jni/%s/%s" %(abi, lib))
    if info:
      extra_path = os.path.join(os.path.dirname(aar), ".unzip")
      if os.path.exists(extra_path):
        shutil.rmtree(extra_path)
      return zip_file.extract(info, extra_path)
    return None

  def run(self, args):
    if args.adb is None:
      adb_path = 'adb'
    else:
      adb_path = args.adb

    if not args.use_su:
      subprocess.check_call([adb_path, "root"])

    try:
      _find_package_pid(adb_path, args.package)
    except (Exception,):
      print "Launch %s first" % args.package
      return 1

    aar_list = self._find_output_aar()
    target = ""
    if len(aar_list) == 0:
      return 1
    elif len(aar_list) == 1:
      target = aar_list[0]
    else:
      choice_prompt = ["- %d %s" % (index, v) for index, v in enumerate(aar_list)]
      prompt = "\r\n".join(choice_prompt) + 'r\nselect index of the aar:'
      index = choice(prompt, choices=["%d" % v for v in range(0, len(aar_list))])
      target = aar_list[int(index)]

    lib_local_path = self._find_lib(target, _get_app_abi(adb_path, args.package), args.lib)
    if not lib_local_path:
      return 1

    lib_tmp_path = _join_path(PushLib.DEVICE_TMP_PATH, args.lib)
    app_lib_path = _join_path(_get_app_path(adb_path, args.package), args.lib)

    AdbShell(adb_path, args.use_su).check_call(["rm", "-f", lib_tmp_path])
    print "1. push %s to device" % args.lib
    subprocess.check_call([adb_path, "push", lib_local_path, lib_tmp_path])
    print "2. killing %s" % args.package
    AdbShell(adb_path).check_call(["am", "force-stop", args.package])
    print "3. copying to %s " % app_lib_path
    AdbShell(adb_path, args.use_su).check_call(["cp", lib_tmp_path, app_lib_path])
    source_file_md5 = md5sum(lib_local_path)
    remote_file_md5 = AdbShell(adb_path, args.use_su).check_output(["md5sum", app_lib_path]).split(" ")[0]
    print "SOURCE FILE MD5: %s\r\nREMOTE FILE MD5: %s" % (source_file_md5, remote_file_md5)
    if source_file_md5 != remote_file_md5:
      print "FAILED!"
      return 1
    print "4. launch %s" % args.package
    AdbShell(adb_path).check_call(["monkey", "-p", args.package, "-c",
      "android.intent.category.LAUNCHER", "1"])
    print "5. DONE!"

    return 0


def main():
  parser = argparse.ArgumentParser("Taobao develop tool")
  subparsers = parser.add_subparsers(help='sub-command help')
  parser.add_argument('--adb', type=str, help="path to ADB tool")
  parser.add_argument('--use_su', dest='use_su', action='store_true')
  parser.add_argument('--no_use_su', dest='use_su', action='store_false')
  parser.set_defaults(use_su=True)

  commands = [
    PushLib(),
  ]
  for command in commands:
    command.add_subparser(subparsers)

  args = parser.parse_args()
  return args.func(args)

if __name__ == '__main__':
  sys.exit(main())
